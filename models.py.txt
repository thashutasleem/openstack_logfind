from djongo import models
from mongoengine import *

class horizon_log(models.Model):
    meta = {'collection': 'horizon_log'}
    id=models.TextField(null=False,primary_key=True)
    remote_host =models.TextField(null=True)
    date = models.DateField(auto_now_add=False,auto_now=False,null=True)
    time = models.TimeField(auto_now=False, auto_now_add=False,null=True)
    zone = models.IntegerField(null=True)
    method = models.TextField(null=True)
    browser_path = models.TextField(null=True)
    apache_status = models.IntegerField(null=True)
    data_transfer = models.IntegerField(null=True)
    info = models.TextField(null=True)
    

    def __str__ (self):
        return self.remote_host
